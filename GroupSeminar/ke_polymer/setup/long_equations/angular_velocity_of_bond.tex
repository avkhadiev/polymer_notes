\label{eq:angular-velocity-of-bond}

  \vec{\omega}_{i}
   &= \frac
        {\vec{r}_{\textrm{bond CM}} \cross \vec{v}_{\textrm{bond CM}}}
        {\vectornorm{\vec{r}_{\textrm{bond CM}}}^2}
\end{aligned}
